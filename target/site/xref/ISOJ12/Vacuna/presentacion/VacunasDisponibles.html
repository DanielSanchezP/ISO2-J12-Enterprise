128     private void iniciarLista(){
129         listVacunas.setModel(modelo);
130         EntregaVacunas entvac= null;
131         List<EntregaVacunas> listaent = gestor.vacunasEnRegion(this.region);
132         for(int i = 0; i < listaent.size();i++){
133             entvac = listaent.get(i);
134             modelo.addElement(entvac.lote.id+" / "+ entvac.lote.farmaceutica + " / " + entvac.cantidad + " / " + entvac.grupoPrioridad + " / " + entvac.fecha);
135         }
136
