156     private void comboBoxListaSitiosActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_comboBoxListaSitiosActionPerformed
157         try {
158             if (comboBoxListaSitios.getSelectedItem().toString().equals("Nacional")){
159                 long total = gestor.consultarTotalVacunados();
160                 totalvacunadosText.setText(Long.toString(total));
161                 double porcentaje = Math.round(gestor.consultarPorcentajeVacunadosSobreRecibidas()*100.0)/100.0;
162                 porcentajevacunadosText.setText(Double.toString(porcentaje));
163             }
164             else{
165                 long totalregion = gestor.consultarTotalVacunadosPorRegion(comboBoxListaSitios.getSelectedItem().toString());
166                 totalvacunadosText.setText(Long.toString(totalregion));
167                 double porcentajeregion = gestor.consultarPorcentajeVacunadosSobreRecibidasEnRegion(comboBoxListaSitios.getSelectedItem().toString());
168                 porcentajevacunadosText.setText(Double.toString(porcentajeregion));
169             }
170         } catch (SQLException ex) {
171                 Logger.getLogger(PantallaGestionSistemaSaludNacional.class.getName()).log(Level.SEVERE, null, ex);
172           }

213     public void inicializarLista(){
214                 lotelist.setModel(modelo);
215                 LoteVacunasDAO lote = new LoteVacunasDAO();
216                 LoteVacunas lotevac= new LoteVacunas();
217        
218         try {
219             
220             List<LoteVacunas> listalote = lote.seleccionarlotes();
221             for(int i = 0; i < listalote.size();i++){
222                 lotevac = listalote.get(i);
223                 modelo.addElement(lotevac.id+" / "+ lotevac.farmaceutica + " / " + lotevac.cantidad);
224             }
225 
226         } catch (SQLException ex) {
227             Logger.getLogger(PantallaGestionSistemaSaludNacional.class.getName()).log(Level.SEVERE, null, ex);
