193     private void botonRegistrarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_botonRegistrarActionPerformed
194         Date fecha = null;
195         SimpleDateFormat formatter = new SimpleDateFormat("dd.MM.yyyy");
196             String dni = dnitext.getText();
197             String nombre = nombrePacienteText.getText();
198             String apellidos = apellidopacientetext.getText();
199             String nombrevacuna = tipovacunatext.getText();
200             int numeroDosis = dosisSlider.getValue();
201             String grupo = grupoComboBox.getSelectedItem().toString();
202             
203         try {
204             fecha = formatter.parse(fechaText.getText());            
205         } catch (ParseException ex) {
206             Logger.getLogger(RegistrarVacunacion.class.getName()).log(Level.SEVERE, null, ex);
207         }
208         gestor.registrarVacunacion(fecha, nombre, apellidos, dni, nombrevacuna, numeroDosis, regionComboBox.getSelectedItem().toString(), grupo);
209         
210     }//GEN-LAST:event_botonRegistrarActionPerformed
