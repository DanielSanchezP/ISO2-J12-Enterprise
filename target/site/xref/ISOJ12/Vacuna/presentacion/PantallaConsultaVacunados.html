122     public void mostrarEstadisticas(){
123         VacunacionDAO vacunacion = new VacunacionDAO();
124         Vacunacion vac= new Vacunacion();
125        
126         try {
127             listaPaciente.setModel(modelo);
128             List<Vacunacion> listavacunacion = vacunacion.seleccionarVacunaciones(this.region);
129             
130             for(int i = 0; i < listavacunacion.size();i++){
131                 vac = listavacunacion.get(i);
132                 modelo.addElement(vac.paciente.nombre+" "+vac.paciente.apellidos);
133             }
134 
135         } catch (SQLException ex) {
136             Logger.getLogger(PantallaConsultaVacunados.class.getName()).log(Level.SEVERE, null, ex);
137         }
138     }
