136     private void botonAltaLoteActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_botonAltaLoteActionPerformed
137         Date fecha = null;
138         SimpleDateFormat formatter = new SimpleDateFormat("dd.MM.yyyy");
139  
140         int cantidad = Integer.parseInt(textoNumVacunas.getText());
141         String farmaceutica = proveedortext.getText();   
142         try {
143             fecha = formatter.parse(textoFecha.getText());
144         } catch (ParseException ex) {
145             Logger.getLogger(RegistrarVacunacion.class.getName()).log(Level.SEVERE, null, ex);
146         }
147         
148         gestor.altaNuevoLoteVacunas(fecha, farmaceutica, cantidad);
149         
150     }//GEN-LAST:event_botonAltaLoteActionPerformed
